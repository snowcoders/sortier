// sortier-ignore-node - Prevents sorting the switch statement along with all nodes inside (e.g. see `a3, a1`)
switch ("a") {
      case "c":
      case "d":
            // The ignore statement stops sorting for the whole tree, not just the node above it
            let {a3, a1} = d;
            console.log("c");
            break;
      case "b":
      case "a":
            console.log("a");
            break;
      default:
            break;
}

// sortier-ignore-line - Doesn't do anything since the switch statement exists across multiple lines
switch ("a") {
      case "d":
      case "c":
            // The ignore statement stops sorting for the whole tree, not just the node above it
            let {a3, a1} = d;
            console.log("c");
            break;
      case "b":
      case "a":
            console.log("a");
            break;
      default:
            break;
}
